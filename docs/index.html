A lightweight ORM for Ignition. SQLAlchemy Core + declarative models + Ignition's system.db.

jython 2.7 sqlalchemy 1.3 ignition 8.x

# Getting started

pip install --target ./site-packages slowrm

See the SQLAlchemy 1.3 docs for full Core and declarative API reference.

# Quick example

from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy import Column, Integer, String, ForeignKey, select
from sqlalchemy.orm import relationship
from slowrm import Session, sync_schema

Base = declarative_base()

class WorkOrder(Base):
    __tablename__ = "work_orders"

    id = Column(Integer, primary_key=True)
    title = Column(String(255))
    status = Column(String(50))
    assignedTo = Column("assigned_to", String(255))

    line_items = relationship("LineItem", cascade="all, delete-orphan")


class LineItem(Base):
    __tablename__ = "line_items"

    id = Column(Integer, primary_key=True)
    work_order_id = Column(Integer, ForeignKey(WorkOrder.id))
    description = Column(String(255))


# Sync schema - creates tables or adds missing columns
sync_schema([WorkOrder, LineItem], "MESDB")

# Insert
with Session("MESDB") as session:
    wo = WorkOrder(title="Replace pump", status="open")
    session.add(wo)
    session.commit()

# Read + Update
with Session("MESDB") as session:
    wo = session.get(WorkOrder, 1)
    wo.status = "complete"
    session.commit()

# Cascade delete
with Session("MESDB") as session:
    wo = session.get(WorkOrder, 1)
    session.delete(wo)
    session.commit()

# Features

Object persistence
add(), get(), delete() with automatic dirty tracking. Mutate objects and commit.
Transaction management
Context manager over Ignition transactions. Commit or rollback as a unit.
Dialect-aware SQL
Auto-detects your datasource type. Generates correct SQL for Postgres, MySQL, MSSQL, SQLite.
Cascade deletes
Define relationships with cascade rules. Children are deleted automatically.
Relationship loading
session.load(wo, WorkOrder.line_items) - explicit eager loading via SQLAlchemy metadata.
Schema sync
sync_schema() creates tables and adds missing columns. No external migration tool needed.

# Session API

MethodDescription
add(instance)
# type: (Base | list) -> None		Queue object(s) for INSERT on commit
merge(instance)
# type: (Base | list) -> Base | list		INSERT if PK not found, UPDATE if exists
get(model, pk, include)
# type: (type, int | tuple, list) -> Base | None		Load by primary key with optional eager loading
delete(instance)
# type: (Base | list) -> None		Queue object(s) for DELETE on commit (cascades if configured)
load(instance, rel)
# type: (Base, attr | str | list) -> None		Eagerly load relationship(s)
query(stmt, params, model, include, as_dict, as_dataset)
# type: (Select, dict, type, list, bool, bool) -> list		Execute SELECT with optional eager loading
query_one(stmt, ...)
# type: (...) -> Base | dict | None		Execute SELECT, return first row or None
execute(stmt, params)
# type: (DML, dict) -> int		Execute INSERT/UPDATE/DELETE directly
execute_many(stmt, param_list)
# type: (DML, list[dict]) -> int		Bulk execute in single DB call (for INSERTs)
flush()
# type: () -> None		Flush pending inserts/updates/deletes
commit()
# type: () -> None		Flush + commit transaction
rollback()
# type: () -> None		Rollback transaction, discard pending changes
compile(stmt, params)
# type: (Select | DML, dict) -> tuple		Return (sql, params) without executing

# Session modes

# Default: transactional, explicit commit
with Session("MESDB") as session:
    session.add(wo)
    session.commit() # required, otherwise rollback on exit

# Autocommit: commits on clean exit
with Session("MESDB", autocommit=True) as session:
    session.add(wo)

# Non-transactional: no tx opened, for read-only queries
with Session("MESDB", transactional=False) as session:
    rows = session.query(stmt, model=WorkOrder)

# Without context manager
session = Session("MESDB")
try:
    wo = session.get(WorkOrder, 42)
    wo.status = "complete"
    session.commit()
except:
    session.rollback()
    raise
finally:
    session.close()

# Merge (upsert)

merge() inserts if the primary key doesn't exist, updates if it does.

from slowrm import Session

# PK exists in DB -> UPDATE
with Session("MESDB") as session:
    wo = session.merge(WorkOrder(id=42, title="Updated", status="complete"))
    session.commit()

# PK doesn't exist in DB -> INSERT
with Session("MESDB") as session:
    wo = session.merge(WorkOrder(id=999, title="Brand new", status="open"))
    session.commit()

# No PK -> INSERT with autoincrement
with Session("MESDB") as session:
    wo = session.merge(WorkOrder(title="Auto ID", status="draft"))
    session.commit()
    print(wo.id) # assigned by DB

# Eager loading

Use include on get(), query(), and query_one() to load relationships in the same session.

# Load a work order with its line items
with Session("MESDB") as session:
    wo = session.get(WorkOrder, 12, include=[WorkOrder.line_items])
    for li in wo.line_items:
        print(li.description)

# Query with eager loading
with Session("MESDB", transactional=False) as session:
    stmt = select([WorkOrder]).where(WorkOrder.status == "open")
    rows = session.query(stmt, model=WorkOrder, include=[WorkOrder.line_items])
    for wo in rows:
        print(wo.title, len(wo.line_items))

# Dirty tracking works on loaded children too
with Session("MESDB") as session:
    wo = session.get(WorkOrder, 12, include=[WorkOrder.line_items])
    wo.line_items[0].description = "Updated"
    session.commit() # flushes UPDATE for the child

# Bulk operations

add(), merge(), and delete() all accept a single instance or a list.

with Session("MESDB") as session:
    # Bulk add
    session.add([
        LineItem(work_order_id=1, description="Part A"),
        LineItem(work_order_id=1, description="Part B"),
        LineItem(work_order_id=1, description="Part C"),
    ])
    session.commit()

with Session("MESDB") as session:
    # Bulk merge (upsert)
    session.merge([
        WorkOrder(id=1, title="Updated A", status="closed"),
        WorkOrder(id=2, title="Updated B", status="closed"),
        WorkOrder(id=99, title="New one", status="open"),
    ])
    session.commit()

with Session("MESDB") as session:
    # Bulk delete
    wo = session.get(WorkOrder, 1, include=[WorkOrder.line_items])
    session.delete(wo.line_items)
    session.commit()

# Relationships

from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy import Column, Integer, String, ForeignKey
from sqlalchemy.orm import relationship
from slowrm import Session, sync_schema

Base = declarative_base()

class Department(Base):
    __tablename__ = "departments"
    id = Column(Integer, primary_key=True)
    name = Column(String(255))
    employees = relationship("Employee", cascade="all, delete-orphan")

class Employee(Base):
    __tablename__ = "employees"
    id = Column(Integer, primary_key=True)
    department_id = Column(Integer, ForeignKey(Department.id))
    name = Column(String(255))

sync_schema([Department, Employee], "MESDB")

# Load relationship
with Session("MESDB") as session:
    dept = session.get(Department, 1)
    session.load(dept, Department.employees)
    for emp in dept.employees:
        print(emp.name)

# Cascade delete - removes department AND all employees
with Session("MESDB") as session:
    dept = session.get(Department, 1)
    session.delete(dept)
    session.commit()

# Composite keys

from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy import Column, Integer, String
from slowrm import Session, sync_schema

Base = declarative_base()

class Tag(Base):
    __tablename__ = "tags"

    entity_type = Column(String(50), primary_key=True)
    entity_id = Column(Integer, primary_key=True)
    tag_name = Column(String(100), primary_key=True)
    tag_value = Column(String(255))

sync_schema([Tag], "MESDB")

# All PK values must be provided (no autoincrement)
with Session("MESDB") as session:
    tag = Tag(entity_type="employee", entity_id=1, tag_name="skill", tag_value="python")
    session.add(tag)
    session.commit()

# Get by composite key (tuple, in definition order)
with Session("MESDB") as session:
    tag = session.get(Tag, ("employee", 1, "skill"))
    tag.tag_value = "jython"
    session.commit()

# Flush

flush() writes pending changes to the database without committing the transaction. Useful when you need a generated ID before adding related objects.

with Session("MESDB") as session:
    # Create parent
    dept = Department(name="Engineering")
    session.add(dept)

    # Flush to get the generated ID
    session.flush()
    print(dept.id) # now has a value

    # Use that ID for children
    session.add(Employee(department_id=dept.id, name="Noah"))
    session.add(Employee(department_id=dept.id, name="Alex"))

    # Commit everything as one transaction
    session.commit()
